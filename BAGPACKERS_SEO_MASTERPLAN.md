import { Metadata } from 'next';

export async function generateMetadata(
  { params }: { params: { slug: string } }
): Promise<Metadata> {
  const caseStudy = await fetchCaseStudyBySlug(params.slug);
  
  if (!caseStudy) {
    return {
      title: 'Case Study Not Found | Bagpackers Developers',
    };
  }

  return {
    title: `${caseStudy.title} | Bagpackers Dev Case Study`,
    description: caseStudy.summary?.slice(0, 155) || 
      `Explore how Bagpackers Developers delivered measurable ROI with ${caseStudy.title}. Custom AI and software engineering case study.`,
    openGraph: {
      title: `${caseStudy.title} | Bagpackers Dev Case Study`,
      description: caseStudy.summary?.slice(0, 155),
      url: `https://bagpackers.dev/case-studies/${params.slug}`,
      type: 'article',
    },
    alternates: {
      canonical: `https://bagpackers.dev/case-studies/${params.slug}`,
    },
  };
}
